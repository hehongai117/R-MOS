```
┌─────────────────────────────────────────────────────────────┐
│                       项目负责人                              │

       │ - 服务层逻辑   │ │ - 组件开发 │ │ - CI/CD       │
```

### 2.2 协作规范

| 规范项 | 要求 |
|--------|------|
| **代码仓库** | 统一使用 `main` 分支，功能分支命名：`feat/模块名-功能描述` |
| **提交规范** | `feat:`, `fix:`, `docs:`, `refactor:`, `test:` 前缀 |
| **Code Review** | 所有 PR 需至少 1 人 Review 后方可合并 |
| **接口文档** | 后端 API 变更需同步更新 Swagger 文档 |
| **每日站会** | 10分钟，同步进度与阻塞问题 |

## 四、接口约定（团队协作关键）

### 4.1 后端 → 前端 API 约定

```yaml
# 任务相关
POST   /api/v1/tasks              # 创建任务（老师）
GET    /api/v1/tasks              # 任务列表（支持 user_id, status 过滤）
GET    /api/v1/tasks/{id}         # 任务详情
POST   /api/v1/tasks/{id}/start   # 开始任务
POST   /api/v1/tasks/{id}/step    # 执行步骤
GET    /api/v1/tasks/{id}/report  # 获取报告（含步骤级扣分）
GET    /api/v1/tasks/{id}/events  # 事件列表

# 用户相关（阶段1新增）
POST   /api/v1/auth/login         # 登录
GET    /api/v1/auth/me            # 当前用户信息
GET    /api/v1/users              # 用户列表（老师可见学生）

# SOP 相关
GET    /api/v1/sops               # SOP 列表
GET    /api/v1/sops/{id}          # SOP 详情（含步骤）

# 故障相关
POST   /api/v1/adapter/inject-fault  # 注入故障
DELETE /api/v1/adapter/fault/{code}  # 清除故障
GET    /api/v1/adapter/faults        # 活动故障列表
```

### 4.2 WebSocket 消息格式

```typescript
// 后端 → 前端
interface WSMessage {
  type: 'ROBOT_STATE' | 'TASK_EVENT' | 'FAULT_ALERT' | 'HEARTBEAT';
  data: {
    joint_states?: JointState[];
    sensor_data?: SensorData;
    active_faults?: string[];
    event?: TaskEvent;
  };
  timestamp: string;
}
```

### 4.3 数据模型约定

```typescript
// 步骤扣分详情（阶段1核心）
interface StepDeduction {
  step_index: number;
  step_title: string;
  max_score: number;
  actual_score: number;
  deduction: number;
  reason: string;
  is_anomaly: boolean;
}

// 用户模型（阶段1新增）
interface User {
  id: number;
  username: string;
  role: 'teacher' | 'student';
  created_at: string;
}
```
